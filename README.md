def display_recipe(category, name):
    """פונקציה המציגה מתכון ספציפי בצורה יפה"""
    recipe = RECIPE_BOOK[category][name]
    print("\n" + "="*40)
    print(f" תפריט: {name} ({category})")
    print(f" זמון הכנה: {recipe['זמן הכנה']} | רמת קושי: {recipe['רמת קושי']}")
    print("="*40)
    
    print("\n🛒 רכיבים:")
    for ing in recipe['רכיבים']:
        print(f"  - {ing}")
        
    print("\n🍳 הוראות הכנה:")
    for i, step in enumerate(recipe['הוראות הכנה'], 1):
        print(f"  {i}. {step}")
    print("="*40 + "\n")

def main():
    """הלולאה המרכזית של אפליקציית המתכונים"""
    print("👋 ברוכים הבאים לספר המתכונים הדיגיטלי הגדול שלי!")
    
    while True:
        print("\nאנא בחרו קטגוריה:")
        categories = list(RECIPE_BOOK.keys())
        for idx, cat in enumerate(categories, 1):
            print(f"{idx}. {cat}")
        print(f"{len(categories) + 1}. יציאה מהתוכנה")
        
        choice = input("\nהקש את מספר הבחירה: ")
        
        if not choice.isdigit():
            print("🚫 קלט לא תקין, נא להקיש מספר בלבד.")
            continue
            
        choice = int(choice)
        
        if choice == len(categories) + 1:
            print("👨‍🍳 בתאבון! נתראה בפעם הבאה.")
            break
            
        if 1 <= choice <= len(categories):
            selected_cat = categories[choice - 1]
            
            while True:
                print(f"\n--- מתכונים בקטגוריית {selected_cat} ---")
                recipes = list(RECIPE_BOOK[selected_cat].keys())
                for idx, rec in enumerate(recipes, 1):
                    print(f"{idx}. {rec}")
                print(f"{len(recipes) + 1}. חזרה לתפריט הראשי")
                
                rec_choice = input("\nבחר מתכון להצגה: ")
                if not rec_choice.isdigit():
                    print("🚫 קלט לא תקין.")
                    continue
                    
                rec_choice = int(rec_choice)
                
                if rec_choice == len(recipes) + 1:
                    break
                    
                if 1 <= rec_choice <= len(recipes):
                    selected_rec = recipes[rec_choice - 1]
                    display_recipe(selected_cat, selected_rec)
                    input("לחץ אנטר (Enter) כדי לחזור לתפריט...")
                else:
                    print("🚫 מספר לא קיים ברשימה.")
        else:
            print("🚫 בחירה לא חוקית, נסה שוב.")

if __name__ == "__main__":
    main()
